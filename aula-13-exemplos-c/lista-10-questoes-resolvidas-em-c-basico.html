10 Questões de Linguagem C com Resolução

Entrada e saída, variáveis e estruturas de decisão.

Questão 1

1
Leia o nome de uma pessoa e mostre: Olá, NOME!
#include <stdio.h>

int main(){

   char nome[30];

   scanf(" %29[^\n]", nome);

   printf("Olá, %s!", nome);

}

Questão 2

2
Leia um número inteiro e mostre o dobro e triplo.
#include <stdio.h>

int main(){

   int n;

   scanf("%d", &n);

   printf("Dobro: %d\nTriplo: %d", n*2, n*3);

}

Questão 3

3
Leia dois números reais e mostre a média.
#include <stdio.h>

int main(){

   float a,b;

   scanf("%f%f", &a, &b);

   printf("Média: %.2f", (a+b)/2);

}

Questão 4

4
Leia nome, idade e salário e exiba os dados.
#include <stdio.h>

int main(){

   char nome[30];
   int idade;
   float salario;

   scanf(" %29[^\n]", nome);
   scanf("%d", &idade);
   scanf("%f", &salario);

   printf("Nome: %s\nIdade: %d\nSalário: %.2f", nome, idade, salario);

}

Questão 7

7
Leia idade, altura e inicial do nome.
#include <stdio.h>

int main(){

   int idade;
   float altura;
   char inicial;

   scanf("%d", &idade);
   scanf("%f", &altura);
   scanf(" %c", &inicial);

   printf("%d %.2f %c", idade, altura, inicial);

}

Questão 10

10
Leia dois números e informe o maior ou empate.
#include <stdio.h>

int main(){

   int a,b;

   scanf("%d%d", &a, &b);

   if(a > b)
      printf("Maior: %d", a);
   else if(b > a)
      printf("Maior: %d", b);
   else
